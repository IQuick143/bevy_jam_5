name = 'Partition 2';

c1 = 'minus_neg';

main = cycle(
    a = vertex(box(c0) button(c1)),
    b = vertex(box(c1) button(c0)),
    vertex(box(c0) button(c0))
    vertex(box(c1) button(c0)),
    c = vertex(box(c0) button(c0)),
    d = vertex(box(c1) button(c0)),
    vertex(box(c0) button(c1))
    vertex(box(c1) button(c1))
    vertex(box(c0) button(c1))
);
